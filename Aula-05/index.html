                import numpy as np
    import pandas as pd

    # Exemplo de dados
    data = {'coluna1': [10, 12, 14, 100, 15, 17, 13, 18, 19, 200, 21, 22, 23],
            'coluna2': [5, 6, 7, 8, 9, 300, 10, 11, 12, 13, 15, 16, 17]}

    # Criação de um DataFrame
    df = pd.DataFrame(data)

    # Função para remover outliers com base no IQR
    def remove_outliers(df):
        for coluna in df.columns:
            Q1 = df[coluna].quantile(0.25)  # Primeiro quartil (Q1)
            Q3 = df[coluna].quantile(0.75)  # Terceiro quartil (Q3)
            IQR = Q3 - Q1  # Intervalo interquartílico (IQR)
            
            # Definindo limites inferior e superior
            limite_inferior = Q1 - 1.5 * IQR
            limite_superior = Q3 + 1.5 * IQR
            
            # Removendo outliers
            df = df[(df[coluna] >= limite_inferior) & (df[coluna] <= limite_superior)]
        
        return df

    # Remover outliers
    df_sem_outliers = remove_outliers(df)

    print("Dados originais:")
    print(df)
    print("\nDados após remoção de outliers:")
    print(df_sem_outliers)
